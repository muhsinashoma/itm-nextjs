5. safe_bool
CREATE OR REPLACE FUNCTION public.safe_bool(v text)
RETURNS boolean AS $func$
BEGIN
    IF v IS NULL OR btrim(v) = '' OR lower(btrim(v)) = 'null' THEN
        RETURN NULL;
    END IF;

    IF lower(btrim(v)) IN ('1', 'true', 't', 'yes', 'y', 'active') THEN
        RETURN TRUE;
    END IF;

    IF lower(btrim(v)) IN ('0', 'false', 'f', 'no', 'n', 'inactive') THEN
        RETURN FALSE;
    END IF;

    RETURN NULL;
END;
$func$ LANGUAGE plpgsql IMMUTABLE;
