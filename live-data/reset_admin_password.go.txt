package main

import (
	"context"
	"fmt"
	"log"
	"time"

	"itm-api/internal/config"
	"itm-api/internal/db"

	"golang.org/x/crypto/bcrypt"
)

func main() {
	cfg := config.Load()

	pool, err := db.Connect(cfg.DatabaseURL)
	if err != nil {
		log.Fatalf("DB connect failed: %v", err)
	}
	defer pool.Close()

	hash, err := bcrypt.GenerateFromPassword([]byte("Admin@1234"), bcrypt.DefaultCost)
	if err != nil {
		log.Fatalf("bcrypt failed: %v", err)
	}

	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
	defer cancel()

	tag, err := pool.Exec(ctx, `
		UPDATE users
		SET password_hash = $1,
		    user_type = 1,
		    active = true
		WHERE username = 'admin'
	`, string(hash))
	if err != nil {
		log.Fatalf("update failed: %v", err)
	}

	fmt.Println("Updated rows:", tag.RowsAffected())
	fmt.Println("Admin username: admin")
	fmt.Println("Admin password: Admin@1234")
}
